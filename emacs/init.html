:ensure t

:config (setq org-agenda-files '("~/Documents/personal/org-mode/")) ;; make org mode allow eval of some langs (org-babel-do-load-languages 'org-babel-load-languages '((emacs-lisp . t) (C . t) (haskell . t) (js . t) (python . t) (scheme . t) (sh . t) (sql . t) (ruby . t)))
